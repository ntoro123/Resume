            import java.util.Random;
            import java.util.Scanner;
            public class guessthenumbergame
            {
                public static void main(String[] args)
                {
                    Scanner scanner = new Scanner(System.in);
                    Random random = new Random();
                    char a;
            
                    do
                    {
                        int x = random.nextInt(1000) + 1;
                        int z = 1;
                        System.out.println("I have a number between 1 and 1000.");
                        System.out.println("Can you guess my number?");
                        System.out.print("Please type your First guess: ");
                        while (true)
                        {
                            int y = scanner.nextInt();
                            if (y == x)
                            {
                                System.out.println("Excellent! You guessed the number in " + z + " attempts."); 
                                break;
                            }
                            else
                            {
                                System.out.println(y < x ? "Too low. Try again." : "Too high. Try again.");
                            }
                            System.out.print("Please type your guess: ");
                            z++;
                            if (y == x)
                            {
                                System.out.println("Excellent! You guessed the number in " + z + " attempts.");
                                break;
                            } 
                            else
                            {
                                System.out.println(y < x ? "Too low. Try again." : "Too high. Try again.");
                            }
                        }
                        System.out.print("Would you like to play again (y or n)? ");
                        a = scanner.next().charAt(0);
                    } while (a == 'y' || a == 'Y');
                }
            }
